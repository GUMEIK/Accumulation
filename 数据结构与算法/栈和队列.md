# 栈
## 封装一个栈结构
```javascript
function Stack(){
    this.arr = [];
    this.push = function(value){
        this.arr.push(value);
    }
    this.pop = function(){
        this.arr.pop();
    }
}

var statck = new Stack();
statck.push(1);
statck.push(2)
console.log(statck.arr)
statck.pop();
console.log(statck.arr)


```

# 队列

## 封装一个队列结构

```javascript
function Queue(){
    this.arr = [];
    this.push = function (value){
        this.arr.push(value);
    }
    this.pop = function(){
        this.arr.shift();
    }
}
var queue = new Queue();
queue.push(1);
queue.push(2)
console.log(queue.arr)
queue.pop();
console.log(queue.arr)
```
