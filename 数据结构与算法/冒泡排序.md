# 排序的本质
排序的本质是 比较 和 交换 。

function compare(a,b){
    if(a > b){
        return true;
    }else{
        return false;
    }
}



function exchange(arr,a,b){
    var temp = arr[a];
    arr[a] = arr[b];
    arr[b] = temp;
}
